    <script>
        // Sélection des éléments
        const menuIcon = document.querySelector('.menu_icon_m');
        const navBar = document.querySelector('.nav_bar');
        const overlay = document.querySelector('.overlay_menu_mobile');
        const closeBtn = document.querySelector('.btn_nav_close');

        // Fonction pour ouvrir le menu
        function openMenu() {
            navBar.classList.add('open');
            overlay.classList.add('open');
            menuIcon.classList.add('active');
            document.body.style.overflow = 'hidden';
        }

        // Fonction pour fermer le menu
        function closeMenu() {
            navBar.classList.remove('open');
            overlay.classList.remove('open');
            menuIcon.classList.remove('active');
            document.body.style.overflow = '';
        }

        // Event listeners
        menuIcon.addEventListener('click', openMenu);
        closeBtn.addEventListener('click', closeMenu);
        overlay.addEventListener('click', closeMenu);

        // Fermer au clic sur un lien (optionnel)
        navBar.querySelectorAll('a').forEach(link => {
            link.addEventListener('click', closeMenu);
        });

        // ===== SOLUTION ROBUSTE POUR LE RESIZE =====
        let resizeTimeout;
        
        function handleResize() {
            // Désactive immédiatement toutes les transitions
            document.body.classList.add('no-transitions');
            
            // Annule le timeout précédent
            clearTimeout(resizeTimeout);
            
            // Ferme le menu si on passe en desktop
            if (window.innerWidth > 860) {
                closeMenu();
            }
            
            // Réactive les transitions après un court délai
            resizeTimeout = setTimeout(() => {
                document.body.classList.remove('no-transitions');
            }, 150);
        }
        
        window.addEventListener('resize', handleResize);
        
        // Gestion du chargement initial pour éviter les flashes
        document.addEventListener('DOMContentLoaded', () => {
            document.body.classList.add('no-transitions');
            setTimeout(() => {
                document.body.classList.remove('no-transitions');
            }, 50);
        });
